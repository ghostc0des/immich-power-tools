# Import Queue Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace the synchronous fire-and-wait import with a SQLite-backed job queue so imports survive page closes and show live progress via polling.

**Architecture:** A job is created in SQLite on request, a background worker processes items concurrently (4 at a time) and writes per-item status back to the db. The frontend polls a status endpoint every 2s. Platform-specific logic is encapsulated in processor classes so Nextcloud/PhotoPrism can be added by dropping in a new file. On server restart, any stuck `processing` jobs are reset to `pending` and restarted when the user's next poll hits the status endpoint.

**Tech Stack:** Drizzle ORM + libsql (SQLite), Next.js API routes, TypeScript, React (polling via `useEffect` + `setInterval`)

// src/workers/import/types.ts
import type { importJobs, importJobItems } from "@/db/schema";

export type ImportJob = typeof importJobs.$inferSelect;
export type ImportJobItem = typeof importJobItems.$inferSelect;

export interface ProcessorContext {
  headers: Record<string, string>;  // user auth headers for Immich API calls
  albumId?: string;                  // resolved after setup()
}

export interface SetupResult {
  skipAssetIds: string[];            // source asset IDs already in Immich → mark skipped
  albumId?: string;                  // created or resolved album ID
  importDataPatch?: Record<string, unknown>; // merged into job.importData in db
}

export interface ImportProcessor {
  /** Called once per job before items are processed. Dedup + album creation. */
  setup(job: ImportJob, context: ProcessorContext): Promise<SetupResult>;
  /** Called once per item. Returns the Immich asset ID on success, throws on failure. */
  processItem(job: ImportJob, item: ImportJobItem, context: ProcessorContext): Promise<{ immichId: string }>;
}
```

import { ENV } from "@/config/environment";
import {
  ensurePowerToolsTag,
  tagAssetWithPowerTools,
  type HeadersRecord,
} from "@/pages/api/import-shared/helpers";
import type { ImportJob, ImportJobItem, ImportProcessor, ProcessorContext, SetupResult } from "../types";

// --- helpers copied/imported from upload-all.ts ---
// (parseFileNameFromDisposition, guessContentType, inferAssetTypeFromName,
//  downloadSharedAsset, uploadAssetBuffer, createImmichAlbum, addAssetToAlbum)
// Copy them here verbatim from upload-all.ts so upload-all.ts stays intact.

const DEVICE_ID = "immich-power-tools";
const makeDeviceAssetId = (assetId: string) => `shared-${assetId}`;

export class ImmichSharedLinkProcessor implements ImportProcessor {
  async setup(job: ImportJob, context: ProcessorContext): Promise<SetupResult> {
    const urlConfig = JSON.parse(job.urlConfig) as { key: string };
    const items = await getJobItems(job.id); // helper defined in runner.ts, passed via closure or import
    const deviceAssetIds = items.map((i) => makeDeviceAssetId(i.assetId));
    const lookup = new Map(items.map((i) => [makeDeviceAssetId(i.assetId).toLowerCase(), i.assetId]));

    const skipAssetIds: string[] = [];
    try {
      const res = await fetch(`${ENV.IMMICH_URL}/api/assets/exist`, {
        method: "POST",
        headers: { ...context.headers, "Content-Type": "application/json" },
        body: JSON.stringify({ deviceAssetIds, deviceId: DEVICE_ID }),
      });
      if (res.ok) {
        const { existingIds = [] } = await res.json();
        for (const id of existingIds) {
          const assetId = lookup.get(id.toLowerCase());
          if (assetId) skipAssetIds.push(assetId);
        }
      }
    } catch (e) {
      console.warn("Dedup check failed, continuing without it", e);
    }

    const importData = JSON.parse(job.importData) as {
      albumOptions?: { createAlbum?: boolean; albumName?: string; addToAlbumId?: string };
      albumId?: string;
    };

    let albumId: string | undefined;
    const albumOptions = importData.albumOptions;
    if (albumOptions?.createAlbum) {
      const name = albumOptions.albumName?.trim() || `Shared import ${new Date().toISOString()}`;
      albumId = await createImmichAlbum(name, {
        ...context.headers,
        "Content-Type": "application/json",
      } as HeadersRecord);
    } else if (albumOptions?.addToAlbumId) {
      albumId = albumOptions.addToAlbumId;
    }

    return { skipAssetIds, albumId, importDataPatch: albumId ? { albumId } : undefined };
  }

  async processItem(job: ImportJob, item: ImportJobItem, context: ProcessorContext): Promise<{ immichId: string }> {
    const urlConfig = JSON.parse(job.urlConfig) as { key: string };
    const itemData = JSON.parse(item.itemData) as {
      originalFileName?: string;
      type: string;
      fileCreatedAt?: string | null;
      localDateTime?: string | null;
      duration?: string | null;
      isFavorite?: boolean;
      isArchived?: boolean;
    };

    const asset = { id: item.assetId, ...itemData };
    const { ["Content-Type"]: _omit, ...uploadHeaders } = context.headers as HeadersRecord;

    const downloaded = await downloadSharedAsset(asset, job.url, urlConfig.key);
    const tag = await ensurePowerToolsTag(context.headers as HeadersRecord);
    const immichId = await uploadAssetBuffer(
      asset,
      downloaded,
      uploadHeaders as HeadersRecord,
      context.headers as HeadersRecord,
      tag.id
    );

    if (context.albumId) {
      await addAssetToAlbum(context.albumId, immichId, {
        ...context.headers,
        "Content-Type": "application/json",
      } as HeadersRecord);
    }

    return { immichId };
  }
}
```

### Task 5: Worker runner

**Files:**
- Create: `src/workers/import/runner.ts`

The runner owns the active-job tracking (in-memory Set), processes items concurrently, and updates the db after each item.

**Step 1: Create the runner**

```typescript
// src/workers/import/runner.ts
import { eq, inArray } from "drizzle-orm";
import { appDb } from "@/db";
import { importJobs, importJobItems } from "@/db/schema";
import { getProcessor } from "./registry";
import "./processors/index"; // side-effect: registers all processors

export type { ImportJob, ImportJobItem } from "./types";

const activeJobs = new Set<string>();
const CONCURRENCY = 4;

export function isJobActive(jobId: string) {
  return activeJobs.has(jobId);
}

export async function getJobItems(jobId: string) {
  return appDb.select().from(importJobItems).where(eq(importJobItems.jobId, jobId));
}

export async function resumePendingJobs() {
  // Reset any jobs stuck in 'processing' from a previous crashed run
  await appDb
    .update(importJobs)
    .set({ status: "pending" })
    .where(eq(importJobs.status, "processing"));
}

export async function runImportJob(
  jobId: string,
  headers: Record<string, string>
): Promise<void> {
  if (activeJobs.has(jobId)) return;
  activeJobs.add(jobId);

  try {
    const [job] = await appDb.select().from(importJobs).where(eq(importJobs.id, jobId));
    if (!job || job.status === "completed" || job.status === "failed") return;

    await appDb.update(importJobs).set({ status: "processing" }).where(eq(importJobs.id, jobId));

    const processor = getProcessor(job.platform);
    const context = { headers };

    // --- Setup: dedup + album creation ---
    const { skipAssetIds, albumId, importDataPatch } = await processor.setup(job, context);

    if (importDataPatch) {
      const merged = { ...JSON.parse(job.importData), ...importDataPatch };
      await appDb.update(importJobs).set({ importData: JSON.stringify(merged) }).where(eq(importJobs.id, jobId));
    }

    if (skipAssetIds.length > 0) {
      await appDb
        .update(importJobItems)
        .set({ status: "skipped" })
        .where(inArray(importJobItems.assetId, skipAssetIds));
      await appDb
        .update(importJobs)
        .set({ skippedCount: skipAssetIds.length })
        .where(eq(importJobs.id, jobId));
    }

    const processorContext = { ...context, albumId };
    const pendingItems = await appDb
      .select()
      .from(importJobItems)
      .where(eq(importJobItems.status, "pending"));

    // --- Concurrent item processing ---
    let idx = 0;
    const worker = async () => {
      while (true) {
        const item = pendingItems[idx++];
        if (!item) break;

        await appDb
          .update(importJobItems)
          .set({ status: "processing" })
          .where(eq(importJobItems.id, item.id));

        try {
          const { immichId } = await processor.processItem(job, item, processorContext);
          await appDb
            .update(importJobItems)
            .set({ status: "uploaded", immichId })
            .where(eq(importJobItems.id, item.id));
          await appDb
            .update(importJobs)
            .set({ uploadedCount: job.uploadedCount + 1 })
            .where(eq(importJobs.id, jobId));
        } catch (err: any) {
          await appDb
            .update(importJobItems)
            .set({ status: "failed", error: err?.message ?? "Unknown error" })
            .where(eq(importJobItems.id, item.id));
          await appDb
            .update(importJobs)
            .set({ failedCount: job.failedCount + 1 })
            .where(eq(importJobs.id, jobId));
        }
      }
    };

    const concurrency = Math.min(CONCURRENCY, pendingItems.length);
    await Promise.all(Array.from({ length: concurrency }, worker));

    await appDb
      .update(importJobs)
      .set({ status: "completed" })
      .where(eq(importJobs.id, jobId));
  } catch (err: any) {
    console.error(`Import job ${jobId} failed fatally:`, err);
    await appDb
      .update(importJobs)
      .set({ status: "failed" })
      .where(eq(importJobs.id, jobId));
  } finally {
    activeJobs.delete(jobId);
  }
}
```

// src/pages/api/import-jobs/index.ts
import type { NextApiRequest, NextApiResponse } from "next";
import { appDb } from "@/db";
import { importJobs, importJobItems } from "@/db/schema";
import { getCurrentUser } from "@/handlers/serverUtils/user.utils";
import { getUserHeaders } from "@/helpers/user.helper";
import { runImportJob } from "@/workers/import/runner";

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method !== "POST") {
    res.setHeader("Allow", "POST");
    return res.status(405).json({ error: "Method Not Allowed" });
  }

  const currentUser = await getCurrentUser(req);
  if (!currentUser) return res.status(401).json({ error: "Unauthorized" });

  const { platform, url, urlConfig, importData, assets } = req.body;

  if (!platform || !url || !Array.isArray(assets) || assets.length === 0) {
    return res.status(400).json({ error: "platform, url, and assets are required" });
  }

  // Create job row
  const [job] = await appDb
    .insert(importJobs)
    .values({
      userId: currentUser.id,
      platform,
      url,
      urlConfig: JSON.stringify(urlConfig ?? {}),
      importData: JSON.stringify(importData ?? {}),
      totalCount: assets.length,
    })
    .returning();

  // Create one item row per asset
  await appDb.insert(importJobItems).values(
    assets.map((asset: { id: string; [key: string]: unknown }) => ({
      jobId: job.id,
      assetId: asset.id,
      itemData: JSON.stringify(asset),
    }))
  );

  // Fire-and-forget: start the worker without awaiting
  const headers = getUserHeaders(currentUser) as Record<string, string>;
  runImportJob(job.id, headers).catch((err) =>
    console.error("Unhandled error in runImportJob", err)
  );

  return res.status(201).json({ jobId: job.id });
}
```

// src/pages/api/import-jobs/[jobId].ts
import type { NextApiRequest, NextApiResponse } from "next";
import { eq } from "drizzle-orm";
import { appDb } from "@/db";
import { importJobs, importJobItems } from "@/db/schema";
import { getCurrentUser } from "@/handlers/serverUtils/user.utils";
import { getUserHeaders } from "@/helpers/user.helper";
import { isJobActive, runImportJob } from "@/workers/import/runner";

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method !== "GET") {
    res.setHeader("Allow", "GET");
    return res.status(405).json({ error: "Method Not Allowed" });
  }

  const currentUser = await getCurrentUser(req);
  if (!currentUser) return res.status(401).json({ error: "Unauthorized" });

  const { jobId } = req.query as { jobId: string };

  const [job] = await appDb.select().from(importJobs).where(eq(importJobs.id, jobId));
  if (!job || job.userId !== currentUser.id) {
    return res.status(404).json({ error: "Job not found" });
  }

  // Auto-resume: if job is pending but has no active worker (e.g. after server restart)
  if (job.status === "pending" && !isJobActive(jobId)) {
    const headers = getUserHeaders(currentUser) as Record<string, string>;
    runImportJob(jobId, headers).catch((err) =>
      console.error("Unhandled error in runImportJob (resume)", err)
    );
  }

  const items = await appDb
    .select()
    .from(importJobItems)
    .where(eq(importJobItems.jobId, jobId));

  return res.status(200).json({ job, items });
}
```
